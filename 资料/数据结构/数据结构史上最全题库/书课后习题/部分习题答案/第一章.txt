int Fibonacci(int k,int m)
{

		for(i=0;i<k+1;i++)              
		
		p[i]=0;
	else if (i=k) p[i]=1;						//第k项为1
	else				                                  //以后的每一项都是前K项的和
		 
		for( i=k+1;i<=m;i++)                                             //分别求从K项到m项的值
		{
		sum+=p[i];     
		p[i]=sum;
		}			

}

typedef enum{A,B,C,D,E} SchoolName;             //定义学校名字	

typedef enum{Female,Male} SexType;              //定义男女

typedef struct{		                            //项目
	
	char event[10];
	
	SexType sex;
	
	SchoolName school;
	
	float score;

}Component;

typedef struct{
	
	float MaleSum;                         //男团总分
	
	float FemaleSum;                       //女团总分
	
	float TotalSum;                        //团体总分

}Sum;


Sum SumScore(SchoolName sn,Component a[],int n)

{
	Sum temp;                                          //赋初值
	
	temp.MaleSum=0;
	
	temp.FemaleSum=0;
	
	temp.TotalSum=0;
	
	int i;
	
	for(i=0;i<n;i++)                                   //循环加总分

	{                                              
		
		if(a[i].school==sn)			//判断是否该学校的
		
		{

		     if(a[i].sex==Male)                      //判断男
				
			  temp.MaleSum+=a[i].score;
			
		     if(a[i].sex==Female)		    //判断女
				
			  temp.FemaleSum+=a[i].score;
		
		}
	
	
	}
	
	temp.TotalSum=temp.MaleSum+temp.FemaleSum;      //总成绩
	
	return temp;
