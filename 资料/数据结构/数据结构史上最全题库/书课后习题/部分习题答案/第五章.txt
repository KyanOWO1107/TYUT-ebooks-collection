#include "stdio.h"
int add(int a,int b)				//定义递归函数
{
	if(b==0) 
		return a;			//实现返回值
	else
		return add(++a,--b);		//a增加，b减少，把b的值给a
}	

void main()
{
	int a,b;
	printf("请输入a,b两个数的值");
	scanf("%d%d",&a,&b);
	printf("%d",add(a,b));			//调用递归函数
}
