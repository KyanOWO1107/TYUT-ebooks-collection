void InsertSort(SqList &L)
{
    int i,j; 
    for(i=2;i<=L.length;++i)
    {
        L.r[L.length+1] = L.r[i];
        for(j=i-1;j>0&&L.r[j].key>L.r[L.length+1].key;--j)
        {
           L.r[j+1] = L.r[j];        //j位为空置位。
        }
        L.r[j+1] = L.r[L.length+1];        
    }
}

void process(int a[n])
{
	low=0;     //low指向开头
	high=n-1;	//high指向结尾
	while(low<high&&a[low]<0)	//判断low小于high并且a[low]小于0
	{
	 low++;			//开头向后移动一个
	while(low<high&&a[high]>0) //判断low小于high并且a[high]大于0
	 high++;
	if(low<high)
		{
		temp=a[low]; //a[low]赋予零时值
		a[low]=a[high];
		a[high]=temp;
		low++;
		high--;
		}
	}
}

void CreateFromTail(LinkList L)  //尾插入法建立单链表
 
{  
 	Node *r, *s; 
 		ElemType x; 
 
		r=L;                //指针动态指向链表的当前表尾，以便于做尾插入，其初值指向头结点 

 		scanf("%d",&x); 
  
	while(x!=-1) 
   		
	{ 
  	s=(Node*)malloc(sizeof(Node));  

   		s->data=x;  
    	
		r->next=s; 
		
		r=s; 
    	
		scanf("%d",&x); 
   
	} 
 
	r->next=NULL;   //将最后一个结点的next链域置为空，表示链表的结束  
 
	}
